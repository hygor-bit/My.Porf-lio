// settle transactions · idempotent & type-safe
export async function settle(batch: Transaction[]) {
  const ledger = await db.transaction(async (tx) => {
    for (const t of batch) {
      await tx.apply(t, { idempotent: true });
    }
    return tx.commit();
  });
  return { status: "settled", count: ledger.size };
}
